レスポンシブiframe埋め込み例

埋め込みコード

<!-- iframe要素 -->
<iframe 
    id="daily-tasks" 
    class="daily-tasks-iframe"
    src="https://your-domain.com/embed.html?d=YOUR_DATA"
    height="400">
</iframe>

<!-- 高さ自動調整スクリプト -->
<script>
window.addEventListener('message', function(e) {
    if (e.data.type === 'setHeight') {
        const iframe = document.getElementById('daily-tasks');
        if (iframe) {
            iframe.style.height = e.data.height + 'px';
        }
    }
});
</script>
